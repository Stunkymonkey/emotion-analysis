\section{Robotics}
\begin{frame}
  \frametitle{Therapy robots for children with autism}
  \begin{itemize}
  \item {\footnotesize \blueemph{Kim, Azzi, et. al (2017): Audio-based emotion estimation for interactive robotic therapy for children with autism spectrum disorder, \textit{14th URAI}. \url{https://ieeexplore.ieee.org/document/7992881}}}
  \item \blueemph{Motivation}: Analyse emotion in speech to interact with children with ASD in real time. Split the reactions between a face and gesture robot, to train recognizing emotions as different expressions together.
  \item \blueemph{Data}: Interactive Emotional Dyadic Motion Capture database; 12h of dialogue; labelled with VAD 
  \item \blueemph{Method}: Feature extraction with openSMILE, 3 SVMs for VAD, real time application: feature normalization with 10 mins of "neutral state" data
  \item \blueemph{Main Result}: Normalization with neutral data improves accuracy for new speakers, dominance has much lower accuracy than valence or activation
  \end{itemize}  
\end{frame}
